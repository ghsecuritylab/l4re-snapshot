299  // special handling for the meta protocol, to avoid 'using' murx
300  template< typename THIS >
301  static l4_msgtag_t _f(THIS *, l4_msgtag_t tag, unsigned r,
302  l4_utcb_t *utcb, True::type)
303  {
304  using L4::Ipc::Msg::dispatch_call;
305  typedef L4::Meta::Rpcs Meta;
306  typedef Meta_svr<IFACE> Msvr;
307  return dispatch_call<Meta>((Msvr *)0, utcb, tag, r);
308  }
309 
310  // normal dispatch to the op_<func> methods of \a self.
311  template< typename THIS >
312  static l4_msgtag_t _f(THIS *self, l4_msgtag_t t, unsigned r,
313  l4_utcb_t *utcb, False::type)
314  {
315  using L4::Ipc::Msg::dispatch_call;
316  return dispatch_call<typename I::iface_type::Rpcs>(self, utcb, t, r);
317  }
318 
319  // dispatch function with switch for meta protocol
320  template< typename THIS >
321  static l4_msgtag_t f(THIS *self, l4_msgtag_t tag, unsigned r,
322  l4_utcb_t *utcb)
323  {
324  if (I::Proto == tag.label())
325  return _f(self, tag, r, utcb, Bool<I::Proto == (long)L4_PROTO_META>());

327  return _Dispatch<IFACE, typename LIST::type>::f(self, tag, r, utcb);
