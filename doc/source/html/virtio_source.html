360 class Request_processor
361 {
362 private:
364  Virtqueue::Desc const *_table;
365 
367  Virtqueue::Desc _current;
368 
370  l4_uint16_t _num;
371 
372 public:
382  template<typename DESC_MAN, typename ...ARGS>
383  void start(DESC_MAN *dm, Virtqueue *ring, Virtqueue::Head_desc const &request, ARGS... args)
384  {
385  _current = cxx::access_once(request.desc());
386 
387  if (_current.flags.indirect())
388  {
389  dm->load_desc(_current, this, &_table);
390  _num = _current.len / sizeof(Virtqueue::Desc);
391  if (L4_UNLIKELY(!_num))
392  throw Bad_descriptor(this, Bad_descriptor::Bad_size);
393 
394  _current = cxx::access_once(_table);
395  }
396  else
397  {
398  _table = ring->desc(0);
399  _num = ring->num();
400  }
401 
402  dm->load_desc(_current, this, cxx::forward<ARGS>(args)...);
403  }
404 
413  template<typename DESC_MAN, typename ...ARGS>
414  Virtqueue::Request const &start(DESC_MAN *dm, Virtqueue::Request const &request, ARGS... args)
415  {
416  start(dm, request.ring, request, cxx::forward<ARGS>(args)...);
417  return request;
418  }
419 
424  Virtqueue::Desc::Flags current_flags() const
425  { return _current.flags; }
426 
431  bool has_more() const
432  { return _current.flags.next(); }
433 
441  template<typename DESC_MAN, typename ...ARGS>
442  bool next(DESC_MAN *dm, ARGS... args)
443  {
444  if (!_current.flags.next())
445  return false;
446 
447  if (L4_UNLIKELY(_current.next >= _num))
448  throw Bad_descriptor(this, Bad_descriptor::Bad_next);
449 
450  _current = cxx::access_once(_table + _current.next);
451 
452  if (0) // we ignore this for performance reasons
453  if (L4_UNLIKELY(_current.flags.indirect()))
454  throw Bad_descriptor(this, Bad_descriptor::Bad_flags);
455 
456  // must throw an exception in case of a bad descriptor
457  dm->load_desc(_current, this, cxx::forward<ARGS>(args)...);
458  return true;
459  }
