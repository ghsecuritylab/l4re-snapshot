340  int add(l4_umword_t match_mask, l4_umword_t match,
341  l4_umword_t del_bits, l4_umword_t add_bits) throw()
342  {
343  l4_msg_regs_t *m = l4_utcb_mr_u(utcb);
344  if (cnt >= L4_UTCB_GENERIC_DATA_SIZE - 4)
345  return -L4_ENOMEM;
346  m->mr[cnt++] = match_mask;
347  m->mr[cnt++] = match;
348  m->mr[cnt++] = del_bits;
349  m->mr[cnt++] = add_bits;
350  return 0;
351  }
