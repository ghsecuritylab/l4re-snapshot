511 L4_INLINE l4_fpage_t
512 __l4_fpage_generic(unsigned long address, unsigned int type,
513  unsigned int size, unsigned char rights) L4_NOTHROW
514 {
515  l4_fpage_t t;
516  t.raw = ((rights << L4_FPAGE_RIGHTS_SHIFT) & L4_FPAGE_RIGHTS_MASK)
517  | ((type << L4_FPAGE_TYPE_SHIFT) & L4_FPAGE_TYPE_MASK)
518  | ((size << L4_FPAGE_SIZE_SHIFT) & L4_FPAGE_SIZE_MASK)
519  | ((address ) & L4_FPAGE_ADDR_MASK);
520  return t;
521 }

583 l4_fpage_contains(l4_fpage_t fpage, l4_addr_t addr, unsigned log2size) L4_NOTHROW
584 {

590 L4_INLINE unsigned char
591 l4_fpage_max_order(unsigned char order, l4_addr_t addr,
592  l4_addr_t min_addr, l4_addr_t max_addr,
593  l4_addr_t hotspot)
594 {
595  while (order < 30 /* limit to 1GB flexpages */)
596  {
597  l4_addr_t mask;
598  l4_addr_t base = l4_trunc_size(addr, order + 1);
599  if (base < min_addr)
600  return order;
601 
602  if (base + (1UL << (order + 1)) - 1 > max_addr - 1)
603  return order;
604 
605  mask = ~(~0UL << (order + 1));
606  if (hotspot == ~0UL || ((addr ^ hotspot) & mask))
607  break;
608 
609  ++order;
610  }
611 
612  return order;
613 }
