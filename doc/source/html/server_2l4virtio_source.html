770  bool setup_queue(Virtqueue *q, unsigned qn, unsigned num_max)
771  {
772  printf("%p:%p: setup queue %d, max %d\n", this, q, qn, num_max);
773  l4virtio_config_queue_t volatile const *qc;
774  qc = _device_config->qconfig(qn);
775 
776  l4_uint64_t pfn = qc->pfn;
777 
778  if (!pfn)
779  {
780  q->disable();
781  return true;
782  }
783 
784  // read to local variables before check
785  l4_uint32_t num = qc->num;
786  l4_uint32_t align = qc->align;
787  l4_uint64_t guest_page_size = _device_config->hdr()->guest_page_size;
788 
789  if (0)
790  printf("%p: setup queue: num=0x%x max_num=0x%x align=0x%x, pfn=0x%x, ps=0x%x\n",
791  this, num, num_max, align, (unsigned)pfn,
792  (unsigned)guest_page_size);
793 
794  if (guest_page_size < 0x1 || guest_page_size > 0x10000000)
795  return false;
796 
797  if (!num || num > num_max)
798  return false;
799 
800  // num must be power of two
801  if (num & (num - 1))
802  return false;
803 
804  // align must be power of two
805  if (align & (align - 1))
806  return false;
807 
808  l4_uint64_t q_addr = pfn * guest_page_size;
809  unsigned long q_size = q->total_size(num, align);
810  auto const *info = _mem_info.find(q_addr, q_size);
811  if (L4_UNLIKELY(!info || !info->is_writable()))
812  return false;
813 
814  printf("shm=[%llx-%llx] local=[%lx-%lx] queue=[%llx-%llx] (%p-%p)\n",
815  info->drv_base(), info->drv_base() + info->size() - 1,
816  (unsigned long)info->local_base(),
817  (unsigned long)info->local_base() + info->size() - 1,
818  q_addr, q_addr + q_size,
819  info->local(Ptr<char>(q_addr)),
820  info->local(Ptr<char>(q_addr)) + q_size);
821  q->setup(num, align, info->local(Ptr<void>(q_addr)));
822  return true;
823  }
