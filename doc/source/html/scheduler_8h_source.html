209 L4_INLINE l4_msgtag_t
210 l4_scheduler_info_u(l4_cap_idx_t scheduler, l4_umword_t *cpu_max,
211  l4_sched_cpu_set_t *cpus, l4_utcb_t *utcb) L4_NOTHROW
212 {
213  l4_msg_regs_t *m = l4_utcb_mr_u(utcb);
214  l4_msgtag_t res;
215 
216  m->mr[0] = L4_SCHEDULER_INFO_OP;
217  m->mr[1] = cpus->gran_offset;
218 
219  res = l4_ipc_call(scheduler, utcb, l4_msgtag(L4_PROTO_SCHEDULER, 2, 0, 0), L4_IPC_NEVER);
220 
221  if (l4_msgtag_has_error(res))
222  return res;
223 
224  cpus->map = m->mr[0];
225 
226  if (cpu_max)
227  *cpu_max = m->mr[1];
228 
229  return res;
230 }

232 L4_INLINE l4_msgtag_t
233 l4_scheduler_run_thread_u(l4_cap_idx_t scheduler, l4_cap_idx_t thread,
234  l4_sched_param_t const *sp, l4_utcb_t *utcb) L4_NOTHROW
235 {
236  l4_msg_regs_t *m = l4_utcb_mr_u(utcb);
237  m->mr[0] = L4_SCHEDULER_RUN_THREAD_OP;
238  m->mr[1] = sp->affinity.gran_offset;
239  m->mr[2] = sp->affinity.map;
240  m->mr[3] = sp->prio;
241  m->mr[4] = sp->quantum;
242  m->mr[5] = l4_map_obj_control(0, 0);
243  m->mr[6] = l4_obj_fpage(thread, 0, L4_FPAGE_RWX).raw;
244 
245  return l4_ipc_call(scheduler, utcb, l4_msgtag(L4_PROTO_SCHEDULER, 5, 1, 0), L4_IPC_NEVER);
246 }

248 L4_INLINE l4_msgtag_t
249 l4_scheduler_idle_time_u(l4_cap_idx_t scheduler, l4_sched_cpu_set_t const *cpus,
250  l4_kernel_clock_t *us, l4_utcb_t *utcb) L4_NOTHROW
251 {
252  l4_msg_regs_t *v = l4_utcb_mr_u(utcb);
253  l4_msgtag_t res;
254 
255  v->mr[0] = L4_SCHEDULER_IDLE_TIME_OP;
256  v->mr[1] = cpus->gran_offset;
257  v->mr[2] = cpus->map;
258 
259  res = l4_ipc_call(scheduler, utcb,
260  l4_msgtag(L4_PROTO_SCHEDULER, 3, 0, 0), L4_IPC_NEVER);
261 
262  if (l4_msgtag_has_error(res))
263  return res;
264 
265  *us = v->mr64[l4_utcb_mr64_idx(0)];
266 
267  return res;
268 }

271 L4_INLINE int
272 l4_scheduler_is_online_u(l4_cap_idx_t scheduler, l4_umword_t cpu,
273  l4_utcb_t *utcb) L4_NOTHROW
274 {
275  l4_sched_cpu_set_t s;
276  l4_msgtag_t r;
277  s.gran_offset = cpu;
278  r = l4_scheduler_info_u(scheduler, NULL, &s, utcb);
279  if (l4_msgtag_has_error(r) || l4_msgtag_label(r) < 0)
280  return 0;
281 
282  return s.map & 1;
283 }
