533 L4_INLINE int
534 l4_sndfpage_add_u(l4_fpage_t const snd_fpage, unsigned long snd_base,
535  l4_msgtag_t *tag, l4_utcb_t *utcb) L4_NOTHROW
536 {
537  l4_msg_regs_t *v = l4_utcb_mr_u(utcb);
538  int i = l4_msgtag_words(*tag) + 2 * l4_msgtag_items(*tag);
539 
540  if (i >= L4_UTCB_GENERIC_DATA_SIZE - 1)
541  return -L4_ENOMEM;
542 
543  v->mr[i] = snd_base | L4_ITEM_MAP | L4_ITEM_CONT;
544  v->mr[i + 1] = snd_fpage.raw;
545 
546  *tag = l4_msgtag(l4_msgtag_label(*tag), l4_msgtag_words(*tag),
547  l4_msgtag_items(*tag) + 1, l4_msgtag_flags(*tag));
548  return 0;
549 }
